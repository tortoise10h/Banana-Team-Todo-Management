create table khoa
  (makhoa int primary key,
   tenkhoa varchar(30) not null);

create table bomon
  (mabm int primary key,
   tenbm varchar(30) not null,
   makhoa int not null references khoa(makhoa));

create table giangvien
  (magv int primary key,
   hoten varchar(30) not null,
   phai varchar(3) check(phai in ('nam', 'nu')),
   tuoi int,
   mabm int not null references bomon(mabm));

create table monhoc
  (mamh int primary key,
   tenmh varchar(30) not null,
   mabm int not null references bomon(mabm));

create table day
  (magv int references giangvien(magv),
   mamh int references monhoc(mamh),
   constraint pk_day primary key(magv, mamh));
